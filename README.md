# How to run:
    - pip install -r requirements.txt
    - mediapipe webcam demo:
        - $python3 webcam-demo.py 
    - mp webcam prediction:
        - $python3 webcam-predict.py
    - CNN Model
        - cnn/cnn-model.ipynb
    - DNN Model x Mediapipe Preprocessing
        - mp/mp-model.ipynb
    - CNN Prediction:
        - $python3 cnn/cnn-predict.py
